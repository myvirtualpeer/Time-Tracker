    <script>

        // REPLACE THIS WITH YOUR GOOGLE APPS SCRIPT WEB APP URL

        const SCRIPT_URL = 'YOUR_GOOGLE_SCRIPT_URL_HERE';

        

        // Update current time display

        function updateTime() {

            const now = new Date();

            const timeString = now.toLocaleString('en-US', {

                weekday: 'long',

                year: 'numeric',

                month: 'long',

                day: 'numeric',

                hour: '2-digit',

                minute: '2-digit',

                second: '2-digit'

            });

            document.getElementById('currentTime').textContent = timeString;

        }

        

        setInterval(updateTime, 1000);

        updateTime();

        

        function recordAction(action) {

            const employeeId = document.getElementById('employeeId').value.trim();

            const statusDiv = document.getElementById('status');

            

            if (!employeeId) {

                statusDiv.className = 'status error';

                statusDiv.textContent = '❌ Please enter your Employee ID';

                return;

            }

            

            const timestamp = new Date().toLocaleString('en-US', {

                year: 'numeric',

                month: '2-digit',

                day: '2-digit',

                hour: '2-digit',

                minute: '2-digit',

                second: '2-digit',

                hour12: true

            });

            

            const data = {

                employeeId: employeeId,

                action: action,

                timestamp: timestamp

            };

            

            // Send to Google Sheets

            if (SCRIPT_URL === 'YOUR_GOOGLE_SCRIPT_URL_HERE') {

                // Demo mode - show what would be sent

                statusDiv.className = 'status success';

                statusDiv.textContent = `✅ ${action} recorded for ID: ${employeeId} at ${timestamp}`;

                console.log('Data to send:', data);

            } else {

                // Send to actual Google Sheets

                fetch(SCRIPT_URL, {

                    method: 'POST',

                    mode: 'no-cors',

                    headers: {

                        'Content-Type': 'application/json',

                    },

                    body: JSON.stringify(data)

                })

                .then(() => {

                    statusDiv.className = 'status success';

                    statusDiv.textContent = `✅ ${action} recorded for ID: ${employeeId}`;

                })

                .catch(error => {

                    statusDiv.className = 'status error';

                    statusDiv.textContent = '❌ Error recording action. Please try again.';

                    console.error('Error:', error);

                });

            }

            

            // Clear status after 3 seconds

            setTimeout(() => {

                statusDiv.style.display = 'none';

            }, 3000);

        }

        

        // Allow Enter key to trigger actions

        document.getElementById('employeeId').addEventListener('keypress', function(e) {

            if (e.key === 'Enter') {

                recordAction('Clock In');

            }

        });
